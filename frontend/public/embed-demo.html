PanPay — ปุ่มจ่ายเงินแบบฝัง (Embed)

ตัวอย่างการฝังหน้าจ่ายเงินลงในเว็บของร้านค้า ด้วยสคริปต์ panpay.js เพียงไฟล์เดียว (popup ไม่ต้องเปลี่ยนหน้า)

1) สร้างรายการชำระเงินจาก Dashboard หรือเรียก POST /v1/charges ฝั่งเซิร์ฟเวอร์ของคุณ เพื่อให้ได้ charge id แล้ววางด้านล่าง:

โค้ดที่ใช้

<script src="http://localhost:8000/panpay.js"></script>
<script>
  PanPay.checkout({
    chargeId: "chg_xxxxxxxxxxxx",         // จาก POST /v1/charges ฝั่งเซิร์ฟเวอร์
    onSuccess: function (e) { alert("ชำระเงินสำเร็จ! ยอด " + e.amount); },
    onClose:   function () { console.log("ปิดหน้าต่าง"); }
  });
</script>
